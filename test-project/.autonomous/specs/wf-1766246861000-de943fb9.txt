FEATURE SPECIFICATION: COUNTER COMPONENT
========================================

OVERVIEW
--------

A minimal, self-contained React counter component built with TypeScript and hooks. The component displays a numeric counter value with three buttons to increment, decrement, and reset the value. The counter supports negative numbers and requires no external dependencies.

REQUIREMENTS
------------

Functional Requirements:
- Display a counter value prominently in the center of the component
- Increment button (+): Adds 1 to the current count
- Decrement button (−): Subtracts 1 from the current count
- Reset button (Reset): Sets the count back to 0
- Counter must support negative values (no minimum bound)
- Counter always starts at 0 (no configurable initial value)

Technical Constraints:
- TypeScript with React functional component
- Use useState hook for state management
- No external dependencies
- No props required
- Inline styles only (no CSS files or modules)
- Named export from barrel file

Component Design:

IMPLEMENTATION STEPS
--------------------

1. Create src/Counter.tsx:

import React, { useState } from 'react';

export const Counter: React.FC = () => {
  const [count, setCount] = useState(0);

  const increment = () => setCount(prev => prev + 1);
  const decrement = () => setCount(prev => prev - 1);
  const reset = () => setCount(0);

  return (
    <div style={{ textAlign: 'center', padding: '20px' }}>
      <div style={{ fontSize: '48px', fontWeight: 'bold', marginBottom: '20px' }}>
        {count}
      </div>
      <div style={{ display: 'flex', gap: '10px', justifyContent: 'center' }}>
        <button onClick={decrement}>−</button>
        <button onClick={increment}>+</button>
        <button onClick={reset}>Reset</button>
      </div>
    </div>
  );
};

2. Create src/Counter.test.tsx:

import React from 'react';
import { render, screen, fireEvent } from '@testing-library/react';
import { Counter } from './Counter';

describe('Counter', () => {
  it('renders with initial value of 0', () => {
    render(<Counter />);
    expect(screen.getByText('0')).toBeInTheDocument();
  });

  it('increments the count when + button is clicked', () => {
    render(<Counter />);
    fireEvent.click(screen.getByText('+'));
    expect(screen.getByText('1')).toBeInTheDocument();
  });

  it('decrements the count when − button is clicked', () => {
    render(<Counter />);
    fireEvent.click(screen.getByText('−'));
    expect(screen.getByText('-1')).toBeInTheDocument();
  });

  it('supports negative values', () => {
    render(<Counter />);
    fireEvent.click(screen.getByText('−'));
    fireEvent.click(screen.getByText('−'));
    fireEvent.click(screen.getByText('−'));
    expect(screen.getByText('-3')).toBeInTheDocument();
  });

  it('resets the count to 0 from a positive value', () => {
    render(<Counter />);
    fireEvent.click(screen.getByText('+'));
    fireEvent.click(screen.getByText('+'));
    fireEvent.click(screen.getByText('Reset'));
    expect(screen.getByText('0')).toBeInTheDocument();
  });

  it('resets the count to 0 from a negative value', () => {
    render(<Counter />);
    fireEvent.click(screen.getByText('−'));
    fireEvent.click(screen.getByText('−'));
    fireEvent.click(screen.getByText('Reset'));
    expect(screen.getByText('0')).toBeInTheDocument();
  });
});
